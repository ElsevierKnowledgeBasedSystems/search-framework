176 std::vector<Instance<State>> makeInstances(int n) {
177  using MyInstance = Instance<State>;
178  std::vector<MyInstance> res;
179  int nStarts = CMD_T.nStarts('w');
180  int nGoals = CMD_T.nGoals('w');
181  std::string strategy = CMD_T.instanceStrategy();
182 
183  // For the random walk strategy.
184  // The next length of random walk is computed by adding walkStep and multiplying by walkMultiplier
185  int walkFirst = CMD_T.walkFirst();
186  int walkRepeat = CMD_T.walkRepeat();
187  int walkIncrement = CMD_T.walkIncrement();
188  int walkMultiplier = CMD_T.walkMultiplier();
189  int curWalkLength = walkFirst;
190  int curRepeated = 0;
191 
192  for (int i = 0; i < n; i++) {
193  std::vector<State> start;
194  std::vector<State> goal;
195  if (i > 0 && strategy == "random_walk")
196  start = res.back().starts();
197  else {
198  for (int i = 0; i != nStarts; i++)
199  start.push_back(uniqueRandomState(start));
200  }
201  for (int i = 0; i != nGoals; i++)
202  if (i == 0 && CMD_T.defaultGoal())
203  goal.push_back(State{});
204  else {
205  if (i > 0 && strategy == "random_walk") {
206  goal.push_back(
207  randomWalkState(goal[0], curWalkLength, goal));
208  if (++curRepeated == walkRepeat) {
209  curWalkLength =
210  (curWalkLength + walkIncrement) * walkMultiplier;
211  curRepeated = 0;
212  }
213  }
214  else
215  goal.push_back(uniqueRandomState(goal));
216  }
217  res.push_back(MyInstance(start, goal));
218  }
219  if (res[0].measures().size()) // otherwise, no measure to sort on
220  std::sort(res.begin(), res.end(), [](const MyInstance &i1,
221  const MyInstance &i2) {
222  return i1.measures()[0].value() < i2.measures()[0].value();
223  });
224  return res;
225 }
