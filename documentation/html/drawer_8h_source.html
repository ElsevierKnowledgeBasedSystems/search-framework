334  double distancePercentile(double p) {
335  assert(p >= 0 && p <= 1.0);
336  std::vector<double> dd;
337  for (auto ed : g_.edgeRange()) {
338  auto from = g_.from(ed), to = g_.to(ed);
339  auto fromPoint = pointMap_.at(from), toPoint = pointMap_.at(to);
340  dd.push_back(gu::distance({fromPoint[0], fromPoint[1]},
341  {toPoint[0], toPoint[1]}));
342  }
343  sort(dd.begin(), dd.end());
344  return dd[p * (dd.size() - 1)];

350  void scaleLayout_(int x, int y) {
351  double minEdgeDistance = distancePercentile(0.001);
352  double vertexSize = minEdgeDistance / 3;
353  std::vector<double> xs, ys;
354  for (auto el : pointMap_) {
355  xs.push_back((el.second)[0]);
356  ys.push_back((el.second)[1]);
357  }
358  double minX = *std::min_element(xs.begin(), xs.end()),
359  maxX = *std::max_element(xs.begin(), xs.end()),
360  minY = *std::min_element(ys.begin(), ys.end()),
361  maxY = *std::max_element(ys.begin(), ys.end());
362 
363  // factors are computed as if currently the vertices ended
364  // exactly at the borders
365  double factorX = x / (maxX - minX + 2 * vertexSize);
366  double factorY = y / (maxY - minY + 2 * vertexSize);
367  double newVertexSize = vertexSize * std::min(factorX, factorY);
368  factorX = (x - 2 * newVertexSize)/ (maxX - minX);
369  factorY = (y - 2 * newVertexSize)/ (maxY - minY);
370 
371  for (auto vd : g_.vertexRange()) {
372  pointMap_.at(vd)[0] =
373  (pointMap_.at(vd)[0] - minX) * factorX + newVertexSize;
374  pointMap_.at(vd)[1] =
375  (pointMap_.at(vd)[1] - minY) * factorY + newVertexSize;
376  }
377  VertexStyle::sizeBase = newVertexSize;
378 
379  double newEdgeWidth = newVertexSize * 2 / 3;
380  EdgeStyle::widthBase = newEdgeWidth;
381  }
