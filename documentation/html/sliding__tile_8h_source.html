223  static AllActions computeAllActions() {
224  AllActions res;
225  for (int blank = 0; blank < size_; ++blank) {
226  // the order is compatible with the code of Richard Korf.
227  if (blank > nColumns - 1)
228  res[blank].push_back(Action{blank - nColumns});
229  if (blank % nColumns > 0)
230  res[blank].push_back(Action{blank - 1});
231  if (blank % nColumns < nColumns - 1)
232  res[blank].push_back(Action{blank + 1});
233  if (blank < size_ - nColumns)
234  res[blank].push_back(Action{blank + nColumns});
235  }
